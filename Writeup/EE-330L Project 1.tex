  \medskip

\section{Input}

\begin{frame}{Input signal and level shifting}
  I wanted to design the FSK pair around transmitting something simple.
  TTL-logic via USB seemed like a good choice.

  \medskip
  There are many common baud rates for serial communication; I initially designed around 9600baud.
\end{frame}

\begin{frame}{Symbol Rate and Bandwidth}
  \begin{equation}
    f_\text{symbol} \leq \frac{1}{10} \cdot \text{Loop Bandwidth}
  \end{equation}

  at 100\,kHz\footnote{couple other factors like maximum catchup/slew rate and also the \ac{PLL} seek time},

  \begin{equation}
    f_\text{symbol} \leq \frac{1}{10} \cdot 100\text{k} = 10\text{kHz} \approx 9600\,\text{baud}
  \end{equation}
\end{frame}

\begin{frame}{PLL modulator}
  $R3$ and $R5$ Set the \textbf{limits} for the pll loop speeds on the (same on demodulator too)

  Reducing the headroom/bandwidth for the PLL max speed helped overall

  Original resistor sizes were specified for the target symbol diversity I wanted.

  \begin{align}
    R_3 &= \qty{47}{\kilo\ohm}  \qquad f_{\mathrm{min}} = \qty{84.4}{\kilo\hertz} \\
    R_5 &= \qty{100}{\kilo\ohm} \qquad f_{\mathrm{min}} = \qty{100.0}{\kilo\hertz}
  \end{align}
\end{frame}
